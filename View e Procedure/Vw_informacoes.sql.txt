create view vw_informacoes 
AS select lote.idlote, lote.fornecedir, dadosentrega.StatusLote, pesquisa.CNPJ from lote
join dadosdaentrega
on lote.idlote = dadosdaentrega.IdLote
join pesquisa 
on pesquisa.idPesquisa = lote.idlote 
order by fonecedor;

select * 
FROM vw_informacoes 
order by fornecedor;
